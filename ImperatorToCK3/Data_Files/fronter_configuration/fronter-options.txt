# Options from this file can be used in the converter's Liquid file templates.

# They can be used in conditions, using the following format:
# {% if <OPTION NAME> == <CHOSEN VALUE> %}
# For example:
# {% if HeresiesInHistoricalAreas == 'yes' %}
# or
# {% if LegionConversion == 'men_at_arms' %}
# or
# {% if ImperatorCivilizationWorth < 0.5 %}
# or
# {% if bookmark_date > '0950-01-01' %}  (needs to be in YYYY-MM-DD format for lexicographic comparisons to work)

# The Liquid condition checks can also be nested, such as:
# {% if RitsuryoRulers == 'japanese_japan' %}
#	some code (A)
# 	{% if FillerGovernments == 'no_changes' %}
# 		some code (B)
# 	{% endif %}
# {% endif %}
# In this case, the code (A) will be added to the file if RitsuryoRulers == 'japanese_japan', and the code (B) will only be added to the file if both FillerGovernments == 'no_changes' AND RitsuryoRulers == 'japanese_japan'

# They can also be used to embed the chosen value in the output, using the following format:
# {{ <OPTION NAME> }}
# For example:
# set_variable = { name = irtock3_civ_worth value = {{ ImperatorCivilizationWorth }} }
# set_variable = { name = irtock3_hist_heresies value = "{{ HeresiesInHistoricalAreas }}" }
# To embed a date value from the converter to check directly in the game's script, such as the bookmark date, it needs to be in this format: {{ bookmark_date | date: 'yyyy.MM.dd' }}

# 2. Add the option to the parser in the RegisterConfigurationKeys method.
