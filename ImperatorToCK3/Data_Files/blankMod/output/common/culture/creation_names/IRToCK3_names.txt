crimean_gothic = {
	trigger = {
		capital_county = {
			title:d_crimea = {
				is_de_jure_liege_or_above_target = prev
			}
		}
		any_sub_realm_county = {
			culture = {
				OR = {
					culture = culture:gothic
					any_parent_culture_or_above = { this = culture:gothic }
				}
			}
			duchy = title:d_crimea
		}
		scope:culture = {
			OR = {
				culture = culture:gothic
				any_parent_culture_or_above = { this = culture:gothic }
			}
		}
	}
}
