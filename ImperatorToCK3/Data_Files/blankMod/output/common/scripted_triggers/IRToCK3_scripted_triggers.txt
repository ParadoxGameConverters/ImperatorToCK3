# This trigger checks if the current holder of a title is in your realm. Intended to be used with counties, but could be used with other tiers too.
# scope = title
is_title_in_realm_trigger = {
	custom_tooltip = {
		text = is_title_in_realm_tt
		holder.top_liege = root
	}
}

# This trigger is used for 'irtock3_split_rome_empire_region_effect' since these exact conditions need to be checked multiple times, so its eaisier to have them handled through a scripted_trigger
# ROOT = Character who ended up causing this effect to trigger (will likely be holding the Rome empire title being split apart)
# scope = empire title being checked
# $REGION$ = The geographical region being checked
irtock3_can_empire_be_used_for_rome_region = {
	# The title must be an empire
	tier = tier_empire
	# Can't have a holder, or the holder is in the same realm as the ROOT ruler (or is the ROOT ruler)
	OR = {
		NOT = { exists = holder }
		holder.top_liege ?= root.top_liege
	}
	# Title must be titular, or have most of its de jure counties already in the REGION
	OR = {
		is_titular = yes
		any_de_jure_county = {
			percent >= 0.75
			title_province = { geographical_region = $REGION$ }
		}
	}
}

# This trigger is used for the Roman prefecture creation decisions.
# scope = hegemony title being checked for if the decision should be shown
# $REGION$ = The geographical region for which the prefecture would be created
irtock3_form_prefecture_is_shown_trigger = {
    OR = {
        has_title = title:h_roman_empire
        has_title = title:h_eastern_roman_empire
        has_title = title:h_western_roman_empire
    }

    # Must hold at least one county in the $REGION$.
    any_realm_county = {
        title_province = { geographical_region = $REGION$ }
    }

    # Check if h_roman_empire, h_western_roman_empire, or h_eastern_roman_empire have an empire vassal with the prefecture_region variable for this region.
    # We don't want the WRE to be able to establish the prefecture of Illyrici if it already exists under the ERE.
    NOR = {
        title:h_roman_empire = {
            any_in_de_jure_hierarchy = {
                tier = tier_empire
                var:prefecture_region ?= geographical_region:$REGION$
                NOT = { is_titular = yes }
            }
        }
        title:h_western_roman_empire = {
            any_in_de_jure_hierarchy = {
                tier = tier_empire
                var:prefecture_region ?= geographical_region:$REGION$
                NOT = { is_titular = yes }
            }
        }
        title:h_eastern_roman_empire = {
            any_in_de_jure_hierarchy = {
                tier = tier_empire
                var:prefecture_region ?= geographical_region:$REGION$
                NOT = { is_titular = yes }
            }
        }
    }
}
