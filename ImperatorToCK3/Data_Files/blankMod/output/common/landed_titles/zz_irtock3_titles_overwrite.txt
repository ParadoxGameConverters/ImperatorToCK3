e_byzantium = {
	color =  hsv{ 0.9 1 0.3 } #hsv{ 0.73 0.8 0.4 }  #hsv{ 0.77 0.8 0.6 } #{ 143 12 125 }

	capital = c_byzantion

	definite_form = yes

	can_create = {
		trigger_if = {
			limit = { exists = title:e_latin_empire.holder }
			rule_title_creation_imperial_power_projection_title_creation_trigger = yes
			# In case the Fourth Crusade content is modified to target either WRE/ERE empire title
			custom_tooltip = {
				text = holds_latin_empire_tt
				NOT = {
					any_held_title = {
						this = title:e_latin_empire
					} 
				}
			}
		}
		trigger_else = {
			custom_tooltip = {
				text = irtock3_title_creation_fourth_crusade_tt
				always = no
			}
		}
	}

	ai_primary_priority = {
		if = {
			limit = {
				culture = { has_cultural_pillar = heritage_byzantine }
			}
			add = @correct_culture_primary_score
		}
	}
}
